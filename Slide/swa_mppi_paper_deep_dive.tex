\section{Motivation \& problem}

\begin{frame}{Why this problem?}
  \begin{itemize}
    \item \textbf{Moral preferences are culturally contingent} (Moral Machine: 40M+ judgments across 233 countries).
    \item \textbf{LLMs} inherit predominantly Western / WEIRD training priors (Henrich et al.).
    \item \textbf{MultiTP} (Jin et al.): multilingual trolley benchmark with country-specific human AMCEs---a rigorous testbed for \emph{alignment to local populations}.
  \end{itemize}
  \vspace{0.3em}
  \begin{block}{Deployment constraint}
    We want \textbf{inference-time} adaptation: \textbf{frozen weights}, \textbf{no per-culture labelled preference data}, \textbf{no reward model training}.
  \end{block}
\end{frame}

\begin{frame}{What existing approaches sacrifice}
  \small
  \begin{tabular}{ll}
    \toprule
    \textbf{Approach} & \textbf{Limitation for cross-cultural moral alignment} \\
    \midrule
    Fine-tuning & Labels + irreversible weight change \\
    Ad-hoc prompting & Weak empirical grounding \\
    Activation steering & White-box, heavy per-model tuning \\
    ARGS / controlled decoding & Trained RM often unavailable \\
    Culture adapters & Scale poorly to 100+ countries \\
    \bottomrule
  \end{tabular}
\end{frame}

\begin{frame}{Core insight: \emph{disagreement is signal}}
  \begin{itemize}
    \item Within-country moral views are \textbf{not monolithic} (age, strata)---supported by survey microdata (WVS Wave~7).
    \item Build \textbf{multiple WVS-grounded personas}; when they \textbf{disagree} on a dilemma, the scenario is \textbf{culturally contested}.
    \item That disagreement triggers a \textbf{bounded, stochastic} correction in \textbf{logit space} (not generation sampling).
  \end{itemize}
\end{frame}

\section{Problem formulation}

\begin{frame}{Setting: binary moral choice in MultiTP}
  \begin{itemize}
    \item Scenario $\mathbf{x}$: forced choice via decision tokens \texttt{LEFT} / \texttt{RIGHT}.
    \item Frozen LLM $f_\theta$: logits $\mathbf{z}=[z_\ell,z_r]$, gap $\Delta=z_r-z_\ell$.
    \item \textbf{Human target}: country AMCE vector $\mathbf{h}^{(c)}\in\mathbb{R}^6$ (six moral dimensions).
    \item \textbf{Model estimate}: $\hat{\mathbf{m}}^{(c)}$ from predicted $p_{\text{spare}}$ pooled per dimension (MultiTP convention; Utilitarianism via regression on count difference).
  \end{itemize}
  \vspace{0.3em}
  \begin{block}{Objective}
    Find logit correction $\delta^\ast(\mathbf{x},c)$ so $\hat{\mathbf{m}}^{(c)}$ is closer to $\mathbf{h}^{(c)}$ under \textbf{Jensen--Shannon divergence} (primary).
  \end{block}
\end{frame}

\section{Method pipeline}

\begin{frame}{Overview: four stages}
  \begin{enumerate}
    \item \textbf{Personas} ($N{=}4$): 3 WVS age cohorts + 1 utilitarian agent (English system prompts; scenario text stays native-language).
    \item \textbf{Batched evaluation}: base + $N$ personas in \textbf{one} forward pass $\Rightarrow$ per-agent decision gaps $\delta_i$.
    \item \textbf{Conflict detection}: if inter-agent variance $\nu\ge\tau^{(c)}$, run MPPI; else use consensus only.
    \item \textbf{PT-weighted MPPI} in logit space + \textbf{positional debiasing} (orig vs.\ lane/group swap).
  \end{enumerate}
\end{frame}

\begin{frame}{Persona construction (WVS $\rightarrow$ natural language)}
  \small
  Eight value dimensions from specific WVS items (e.g.\ gender egalitarianism Q58P--Q60P, religion Q6P, trust Q43P, \ldots).
  \begin{itemize}
    \item Segment respondents into \textbf{young / middle / older}; map dimension means to \textbf{natural-language descriptors}.
    \item Fourth persona: \textbf{utilitarian} (save more lives)---ablations show it is load-bearing for ranking alignment.
  \end{itemize}
  \vspace{0.2em}
  \begin{alertblock}{Design principle}
    Personas reflect \textbf{measured population structure}, not author stereotypes (mitigates toxicity risks from naive cultural prompts).
  \end{alertblock}
\end{frame}

\begin{frame}{Batched forward pass \& gaps}
  \[
    \mathbf{z}_{\text{base}},\mathbf{z}_1,\ldots,\mathbf{z}_N
    = f_\theta(\mathbf{x};\varnothing,\mathbf{s}_1,\ldots,\mathbf{s}_N)
  \]
  Per-category logit temperatures $T_{\text{cat}}$ scale logits before gaps:
  \[
    \delta_i=\frac{z_{i,r}-z_{i,\ell}}{T_{\text{cat}}},\quad
    \bar\delta=\frac1N\sum_i\delta_i,\quad
    r_i=\delta_i-\delta_{\text{base}}.
  \]
  \textbf{Reward} $r_i$: how strongly persona $i$ pulls away from the neutral baseline on this scenario.
\end{frame}

\begin{frame}{Adaptive threshold $\tau^{(c)}$ (when to intervene)}
  \begin{itemize}
    \item Calibrate on \textbf{unlabelled} scenarios using only \textbf{inter-agent variance} statistics---\textbf{no} human AMCE access.
    \item Target $\approx$35\% MPPI trigger rate (percentile on calibration variances); main paper uses \textbf{20\%/80\% held-out split} for $\tau$ vs.\ evaluation.
    \item Robustness: held-out vs.\ in-distribution calibration yields $|\Delta\mathrm{JSD}|=0.0006$ (below bootstrap CI $\pm 0.004$).
  \end{itemize}
\end{frame}

\begin{frame}{Prospect Theory utility (per-agent)}
  Cooperative decomposition with $\lambda_{\text{coop}}=0.7$:
  \[
    U_i(\epsilon_k)=
    (1-\lambda_{\text{coop}})\,v(r_i\tilde\delta_k)
    +\lambda_{\text{coop}}\,v(\bar r_{-i}\,\tilde\delta_k),\quad
    \tilde\delta_k=\bar\delta+\epsilon_k.
  \]
  Classical PT value:
  \[
    v(x)=\begin{cases}x^\alpha & x\ge 0\\ -\kappa|x|^\beta & x<0\end{cases}
    \quad (\alpha=\beta=0.88,\;\kappa=2.25).
  \]
\end{frame}

\begin{frame}{Collective objective + MPPI weights}
  \[
    U_{\text{total}}(\epsilon_k)=\frac1N\sum_i U_i(\epsilon_k)
    -\alpha_{\text{KL}}\frac{\epsilon_k^2}{2\sigma^2}
    \quad(\alpha_{\text{KL}}=0.05).
  \]
  \[
    w_k=\frac{\exp(U_{\text{total}}(\epsilon_k)/\beta)}{\sum_{k'}\exp(\ldots)},\qquad
    \delta^\ast=\sum_k w_k\epsilon_k,\quad \epsilon_k\sim\mathcal N(0,\sigma^2).
  \]
  \textbf{Key design}: $\beta=T_{\text{dec}}=0.5$ aligns MPPI smoothing scale with the final sigmoid decision temperature.
\end{frame}

\begin{frame}{Positional debiasing}
  \begin{itemize}
    \item Evaluate each scenario twice: \textbf{original} vs.\ \textbf{swap} (LEFT$\leftrightarrow$RIGHT lanes + Group A$\leftrightarrow$B in native script).
    \item Debiased gap: $\delta_i=\big(\delta_i^{\text{orig}}-\delta_i^{\text{swap}}\big)/2$ (mitigates recency / side bias).
    \item Final: $p_{\text{spare}}=\sigma\big((\bar\delta+\delta^\ast)/T_{\text{dec}}\big)$.
  \end{itemize}
\end{frame}

\section{Experiments}

\begin{frame}{Benchmark, languages, preprocessing}
  \begin{itemize}
    \item \textbf{MultiTP}: 15 countries, native-language vignettes, 6 moral dimensions (Species, Gender, Age, Fitness, Social Value, Utilitarianism).
    \item \textbf{12 languages} for scenarios; \textbf{English} \texttt{LEFT}/\texttt{RIGHT} tokens verified as single-token in each model vocabulary.
    \item Preprocessing: dedup, quality filter (Utilitarianism edge cases), per-category cap, upsampling for stable AMCE---with \textbf{sensitivity suite} over configurations.
  \end{itemize}
\end{frame}

\section{Mechanisms \& robustness}

\begin{frame}{What creates the gain? (CS + Binary-ARGS)}
  \small
  \begin{itemize}
    \item \textbf{Consensus-shift baselines} on $\bar\delta$: best is CS-Clamp; SWA-MPPI still wins by $\sim 0.003$ JSD $\Rightarrow$ residual explained by \textbf{PT asymmetry}, not WVS weighting alone.
    \item \textbf{Binary-ARGS}: ARGS-PT matches SWA-MPPI kernel; ARGS-WVS can \emph{hurt} vs.\ uniform without PT asymmetry---isolates \textbf{Prospect Theory} as the mechanism.
  \end{itemize}
\end{frame}

\begin{frame}{``Anglosphere paradox''}
  \begin{itemize}
    \item Largest JSD improvements often in \textbf{English-speaking} countries (e.g.\ AU/GB/US up to $\sim$+72--75\% JSD improvement on Qwen2.5-72B).
    \item Interpretation: models can be \textbf{misaligned with local human AMCEs} even when English data is over-represented---RLHF-era priors do not guarantee country-level moral calibration.
  \end{itemize}
\end{frame}

\begin{frame}{Robustness (high level)}
  \footnotesize
  \begin{itemize}
    \item \textbf{$\tau$}: held-out calibration vs.\ leakage---negligible aggregate JSD gap.
    \item \textbf{Temperatures}: three sweeps, JSD spans $<0.010$; defaults conservative vs.\ best point.
    \item \textbf{Dataset preprocessing}: six configs, aggregate JSD range $0.0034$.
    \item \textbf{Qwen2.5-32B}: entropy-aware $\sigma$ confirms concentrated logits; explains MPPI degradation.
  \end{itemize}
\end{frame}

\section{Takeaways}

\begin{frame}{Contributions (summary)}
  \begin{enumerate}
    \item First \textbf{MPPI in logit space} for cross-cultural moral alignment, training-free.
    \item Strong empirical gains on \textbf{70B-class} models (+34--38\% mean JSD), evaluated across \textbf{six} architectures; smaller models +18--23\%.
    \item Clear \textbf{ablation hierarchy} (utilitarian $>$ MPPI $>$ debias $>$ PT).
    \item Extensive \textbf{robustness + mechanistic} evidence ($\tau$, temps, data, CS, ARGS, entropy diagnostic).
  \end{enumerate}
\end{frame}

\begin{frame}{Limitations (honest)}
  \small
  \begin{itemize}
    \item Binary forced-choice / single decision position---extension to multi-option or open-ended is non-trivial.
    \item WVS personas are a proxy; sparse WVS coverage hurts some countries.
    \item Hyperparameter validation used synthetic author-annotated proxy labels + transfer checks---real-data validation remains valuable.
  \end{itemize}
\end{frame}
